#+title: Java 14 Switch Yield

* Requirements:

  - Java 14

* Code

  #+begin_src java
    public class Foo {

      public static void main(String[] args) {
        int q = 1;
        String result = switch (q) {
          case 1 -> "foo";
          case 2 -> {
            yield "bar";
          }
          default -> "nope";
        };
        System.out.println(result);
      }

    }
  #+end_src
